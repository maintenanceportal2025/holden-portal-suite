    <script>
        // Email Service Configuration
        const EMAIL_SERVICE_URL = 'https://script.google.com/macros/s/AKfycbwwSy44WvoHQIKiUn9Ctf0rHyYCuLPNFBrlSVSexks7rVnRvdVQeZC9F3Zwu2RruX3v/exec';
        
        // Global state
        let emailTemplates = [];
        let selectedTemplate = null;
        let currentRecipients = [];

        // Initialize on page load
        document.addEventListener('DOMContentLoaded', function() {
            initializeMassNotification();
        });

        // Initialize the mass notification system
        async function initializeMassNotification() {
            try {
                showLoading(true);
                
                // Check email service status
                await checkServiceStatus();
                
                // Load email templates
                await loadEmailTemplates();
                
                // Setup event listeners
                setupEventListeners();
                
                showLoading(false);
                
            } catch (error) {
                console.error('‚ùå Error initializing mass notification:', error);
                showError('Failed to initialize mass notification system: ' + error.message);
                showLoading(false);
            }
        }

        // Check email service status
        async function checkServiceStatus() {
            try {
                const response = await makeServiceRequest('getEmailConfig');
                
                if (response.success) {
                    updateServiceStatus(true);
                } else {
                    throw new Error('Service not responding');
                }
                
            } catch (error) {
                console.error('‚ùå Email service offline:', error);
                updateServiceStatus(false);
            }
        }

        // Update service status display
        function updateServiceStatus(isOnline) {
            const indicator = document.getElementById('serviceIndicator');
            const status = document.getElementById('serviceStatus');
            
            if (isOnline) {
                indicator.className = 'status-indicator';
                status.textContent = 'Online';
            } else {
                indicator.className = 'status-indicator offline';
                status.textContent = 'Offline';
            }
        }

        // Load email templates
        async function loadEmailTemplates() {
            try {
                const response = await makeServiceRequest('getEmailTemplates');
                
                if (response.success) {
                    emailTemplates = response.templates;
                    renderTemplateGrid();
                    document.getElementById('templateCount').textContent = emailTemplates.length;
                } else {
                    throw new Error(response.error || 'Failed to load templates');
                }
                
            } catch (error) {
                console.error('‚ùå Error loading templates:', error);
                showError('Failed to load email templates: ' + error.message);
            }
        }

        // Render template selection grid
        function renderTemplateGrid() {
            const container = document.getElementById('templateGrid');
            
            if (emailTemplates.length === 0) {
                container.innerHTML = '<p style="color: #64748b; font-style: italic;">No templates available</p>';
                return;
            }
            
            container.innerHTML = emailTemplates.map(template => `
                <div class="template-card" onclick="selectTemplate('${template.templateId}')">
                    <div class="template-category">${template.category}</div>
                    <div class="template-title">${template.templateId}</div>
                    <div class="template-preview">${template.subject}</div>
                </div>
            `).join('');
        }

        // Select a template
        function selectTemplate(templateId) {
            // Remove previous selection
            document.querySelectorAll('.template-card').forEach(card => {
                card.classList.remove('selected');
            });
            
            // Find and select template
            selectedTemplate = emailTemplates.find(t => t.templateId === templateId);
            
            if (selectedTemplate) {
                // Visual selection
                event.target.closest('.template-card').classList.add('selected');
                
                // Show variable inputs
                renderVariableInputs();
                
                // Update preview
                updatePreview();
                
                // Enable send button
                document.getElementById('sendEmailBtn').disabled = false;
            }
        }

        // Render variable input fields
        function renderVariableInputs() {
            const container = document.getElementById('variableInputs');
            
            if (!selectedTemplate || !selectedTemplate.variables) {
                container.innerHTML = '<p style="color: #64748b; font-style: italic;">No customization needed for this template</p>';
                return;
            }
            
            const variables = selectedTemplate.variables.split(',');
            
            container.innerHTML = variables.map(variable => {
                const varName = variable.trim();
                return `
                    <div class="form-group">
                        <label class="form-label">${varName}</label>
                        <input type="text" class="form-input" id="var_${varName}" 
                               placeholder="Enter ${varName.toLowerCase()}" 
                               onchange="updatePreview()">
                    </div>
                `;
            }).join('');
        }

        // Update email preview
        function updatePreview() {
            if (!selectedTemplate) {
                document.getElementById('previewPanel').style.display = 'none';
                document.getElementById('recipientInfo').style.display = 'none';
                return;
            }
            
            // Get variable values
            const variables = {};
            if (selectedTemplate.variables) {
                selectedTemplate.variables.split(',').forEach(variable => {
                    const varName = variable.trim();
                    const input = document.getElementById('var_' + varName);
                    variables[varName] = input ? input.value || `[${varName}]` : `[${varName}]`;
                });
            }
            
            // Generate preview content
            let subject = selectedTemplate.subject;
            let body = selectedTemplate.body;
            
            // Replace variables
            Object.keys(variables).forEach(varName => {
                const placeholder = `{${varName}}`;
                subject = subject.replace(new RegExp(placeholder, 'g'), variables[varName]);
                body = body.replace(new RegExp(placeholder, 'g'), variables[varName]);
            });
            
            // Replace common variables
            const today = new Date().toLocaleDateString('en-AU');
            subject = subject.replace(/{DATE}/g, today);
            body = body.replace(/{DATE}/g, today);
            
            // Update preview
            document.getElementById('previewSubject').textContent = subject;
            document.getElementById('previewBody').textContent = body;
            document.getElementById('previewPanel').style.display = 'block';
            
            // Update recipient info
            updateRecipientInfo();
        }

        // Update recipient information
        function updateRecipientInfo() {
            const filter = document.getElementById('recipientFilter').value;
            
            // Estimate recipient counts
            const counts = {
                'all_units': { count: 44, description: 'All retirement village units' },
                'zone_1': { count: 8, description: 'Zone 1 units only' },
                'zone_2': { count: 9, description: 'Zone 2 units only' },
                'zone_3': { count: 9, description: 'Zone 3 units only' },
                'zone_4': { count: 9, description: 'Zone 4 units only' },
                'zone_5': { count: 9, description: 'Zone 5 units only' }
            };
            
            const info = counts[filter] || counts['all_units'];
            
            document.getElementById('recipientCount').textContent = `${info.count} recipients`;
            document.getElementById('recipientDetails').textContent = info.description;
            document.getElementById('recipientInfo').style.display = 'block';
        }

        // Send mass notification
        async function sendMassNotification() {
            if (!selectedTemplate) {
                showError('Please select a template first');
                return;
            }
            
            try {
                showLoading(true);
                
                // Get variable values
                const customData = {};
                if (selectedTemplate.variables) {
                    selectedTemplate.variables.split(',').forEach(variable => {
                        const varName = variable.trim();
                        const input = document.getElementById('var_' + varName);
                        if (input) {
                            customData[varName] = input.value || '';
                        }
                    });
                }
                
                const recipientFilter = document.getElementById('recipientFilter').value;
                
                // Call email service
                const response = await makeServiceRequest('sendMassNotification', {
                    templateId: selectedTemplate.templateId,
                    recipientFilter: recipientFilter,
                    customData: JSON.stringify(customData),
                    adminUser: 'MassNotificationInterface'
                });
                
                if (response.success) {
                    showSuccess(`‚úÖ Email sent successfully to ${response.recipientCount} recipients!`);
                    
                    // Reset form
                    setTimeout(() => {
                        resetForm();
                    }, 2000);
                } else {
                    throw new Error(response.error || 'Failed to send mass notification');
                }
                
            } catch (error) {
                console.error('‚ùå Error sending mass notification:', error);
                showError('Failed to send email: ' + error.message);
            } finally {
                showLoading(false);
            }
        }

        // Setup event listeners
        function setupEventListeners() {
            // Recipient filter change
            document.getElementById('recipientFilter').addEventListener('change', updateRecipientInfo);
        }

        // Reset form
        function resetForm() {
            selectedTemplate = null;
            document.querySelectorAll('.template-card').forEach(card => {
                card.classList.remove('selected');
            });
            document.getElementById('variableInputs').innerHTML = '<p style="color: #64748b; font-style: italic;">Select a template to see customization options</p>';
            document.getElementById('previewPanel').style.display = 'none';
            document.getElementById('recipientInfo').style.display = 'none';
            document.getElementById('sendEmailBtn').disabled = true;
        }

        // Refresh preview
        function refreshPreview() {
            if (selectedTemplate) {
                updatePreview();
                showSuccess('Preview refreshed');
            }
        }

        // Make request to email service
        async function makeServiceRequest(action, params = {}) {
            const requestParams = {
                action: action,
                ...params
            };
            
            const url = EMAIL_SERVICE_URL + '?' + new URLSearchParams(requestParams).toString();
            
            return new Promise((resolve, reject) => {
                const script = document.createElement('script');
                const callbackName = 'emailServiceCallback_' + Date.now();
                
                window[callbackName] = function(data) {
                    delete window[callbackName];
                    document.body.removeChild(script);
                    resolve(data);
                };
                
                script.src = url + '&callback=' + callbackName;
                script.onerror = () => {
                    delete window[callbackName];
                    document.body.removeChild(script);
                    reject(new Error('Failed to connect to email service'));
                };
                
                document.body.appendChild(script);
                
                setTimeout(() => {
                    if (window[callbackName]) {
                        delete window[callbackName];
                        document.body.removeChild(script);
                        reject(new Error('Email service request timeout'));
                    }
                }, 10000);
            });
        }

        // Utility functions
        function showLoading(show) {
            document.body.classList.toggle('loading', show);
        }

        function showSuccess(message) {
            showNotification(message, 'success');
        }

        function showError(message) {
            showNotification(message, 'error');
        }

        function showNotification(message, type) {
            const notification = document.createElement('div');
            notification.className = `notification ${type}`;
            notification.textContent = message;
            
            document.body.appendChild(notification);
            
            setTimeout(() => {
                notification.classList.add('show');
            }, 100);
            
            setTimeout(() => {
                notification.classList.remove('show');
                setTimeout(() => {
                    document.body.removeChild(notification);
                }, 300);
            }, 4000);
        }
    </script>
